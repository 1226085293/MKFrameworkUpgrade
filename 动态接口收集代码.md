const prototype_tabs: Record<string, string>[] = [];
const global_tab: Record<string, string> = {};

const convert_name_f = (
	name_s_: string,
	first_upper_case_b = false
): string => {
	const content_match = name_s_.match(
		/(?<=([_]*))([A-Za-z\d]{1})([A-Za-z\d_]+)([A-Za-z\d]{1})(?=([_]*))/
	)!;

	if (!content_match) {
		return name_s_;
	}

	const end_match = content_match[0].match(/_(n|b|s|a)([s]*)$/g);
	let new_name_s = content_match[0];

	if (end_match) {
		new_name_s = new_name_s.slice(0, -end_match[0].length);

		if (end_match[0].startsWith("_n")) {
			new_name_s += "Num";
		} else if (end_match[0].startsWith("_s")) {
			new_name_s += "Str";
		} else if (end_match[0] === "_b") {
			new_name_s = "is" + new_name_s[0].toUpperCase() + new_name_s.slice(1);
		} else if (end_match[0].startsWith("_f")) {
			new_name_s += "Func";
		}

		if (end_match[0] === "_as" || end_match[0].length > 3) {
			new_name_s += "List";
		}
	}

	new_name_s = new_name_s.replace(/_[\w]/g, function (...args_as: any[]) {
		return (args_as[0] as string).slice(1).toUpperCase();
	});

	if (first_upper_case_b) {
		new_name_s = new_name_s[0].toUpperCase() + new_name_s.slice(1);
	}

	return name_s_
		.replace(content_match[0], new_name_s)
		.replace(/^currTimeSNum$/, "currentTimeSNum");
};

const get_all_property = (obj_: any): string[] => {
	const props = new Set();
	const object_proto_props = new Set(
		Object.getOwnPropertyNames(Object.prototype)
	);
	let current = obj_;

	while (current !== null && current !== Object.prototype) {
		Object.getOwnPropertyNames(current).forEach((p) => {
			if (!object_proto_props.has(p) && isNaN(Number(p))) {
				// 排除 Object.prototype 的原生属性
				props.add(p);
			}
		});

		current = Object.getPrototypeOf(current);
	}

	return Array.from(props) as any;
};

const find_prototype_f = (
	start_s_: string,
	object_: any,
	key_s_: string,
	enum_b = false
): void => {
	if (
		(key_s_ && !isNaN(Number(key_s_))) ||
		object_ === cc ||
		object_ instanceof cc.Node ||
		object_ instanceof cc.Component ||
		object_ instanceof cc.EventTarget
	) {
		return;
	}

	const index_n = start_s_.match(/\./g)?.length ?? 0;
	let previous_start_s = !index_n
		? start_s_
		: prototype_tabs[index_n - 1][start_s_.slice(0, -key_s_.length - 1)] + ".";

	previous_start_s = previous_start_s.replace("global_config", "GlobalConfig");
	if (!prototype_tabs[index_n]) {
		prototype_tabs[index_n] = {};
	}

	switch (typeof object_) {
		case "function": {
			// class
			if ((object_.toString() as string).startsWith("class")) {
				prototype_tabs[index_n][
					start_s_
				] = `${previous_start_s}${convert_name_f(key_s_, true)}`;

				const key_ss = get_all_property(object_.prototype);
				const prototype_ss: string[] =
					object_.toString().match(/(?<=this\.)([\w_]+)(?= = )/g) ?? [];

				[...key_ss, ...prototype_ss.filter((v_s) => v_s[0] !== "_")].forEach(
					(v_s) => {
						let type_s = "";

						try {
							type_s = typeof object_.prototype[v_s];
						} catch (e) {
							//
						}

						if (type_s === "function" && v_s[0] === "_") {
							return;
						}

						if (convert_name_f(v_s) !== v_s) {
							global_tab[v_s] = convert_name_f(v_s);
						}
					}
				);

				prototype_tabs[index_n][
					start_s_
				] = `${previous_start_s}${convert_name_f(key_s_, true)}`;
			}
			// 函数
			else {
				prototype_tabs[index_n][
					start_s_
				] = `${previous_start_s}${convert_name_f(key_s_)}`;
			}

			break;
		}

		case "object": {
			if (!object_) {
				return;
			}

			const property_ss = get_all_property(object_);
			let enum_b = false;

			// 命名空间
			if (
				(!key_s_.startsWith("_") && key_s_.endsWith("_")) ||
				(start_s_.startsWith("global_config") &&
					(start_s_.match(/\./g)?.length ?? 0) <= 1)
			) {
				prototype_tabs[index_n][
					start_s_
				] = `${previous_start_s}${convert_name_f(key_s_, true)}`;
			}
			// 对象
			else {
				enum_b =
					object_["proxySymbol635461"] !== "proxySymbol635461" &&
					(object_.__enums__ ||
						Object.keys(object_).length === property_ss.length * 2);

				// 如果对象是枚举
				if (enum_b) {
					prototype_tabs[index_n][
						start_s_
					] = `${previous_start_s}${convert_name_f(key_s_, true)}`;
				}
				// 正常对象
				else {
					prototype_tabs[index_n][
						start_s_
					] = `${previous_start_s}${convert_name_f(key_s_)}`;
				}
			}

			get_all_property(object_).forEach((v_s) => {
				find_prototype_f(`${start_s_}.${v_s}`, object_[v_s], v_s, enum_b);
			});

			break;
		}

		case "string":
		case "number":
		case "boolean": {
			if (start_s_ === `${previous_start_s}${convert_name_f(key_s_, enum_b)}`) {
				return;
			}

			prototype_tabs[index_n][start_s_] = `${previous_start_s}${convert_name_f(
				key_s_,
				enum_b
			)}`;
			break;
		}
	}
};
